00911 // builds an equivalent Expression_Tree node
00912 Madara::Expression_Tree::Component_Node *
00913 Madara::Expression_Tree::Both::build (void)
00914 {
00915   // Since users can say something like ";;;;;;;;", it is very possible
00916   // that a both operation contains no valid children. So, we need
00917   // to check whether or not we have a valid child.
00918   if (left_ && right_)
00919     return new Composite_Both_Node (left_->build (), right_->build ());
00920   else if (left_)
00921     // all we have is a valid left child, so there is no reason to build
00922     // a Both operator
00923     return left_->build ();
00924   else if (right_)
00925     // all we have is a valid right child, so there is no reason to build
00926     // a Both operator
00927     return right_->build ();
00928   else
00929     // we've got nothing. This node should eventually be pruned out of the
00930     // picture if at all possible.
00931     return new Leaf_Node ((long long)0);
00932 }

01282 // inserts a variable (leaf node / number) into the parse tree
01283 void
01284 Madara::Expression_Tree::Interpreter::variable_insert (Madara::Knowledge_Engine::Thread_Safe_Context &context,
01285                                                        const ::std::string &input,
01286                                                        ::std::string::size_type &i,
01287                                                        int & accumulated_precedence,
01288                                                        ::std::list<Symbol *>& list,
01289                                                        Symbol *& lastValidInput)
01290 {
01291   // merge all consecutive number chars into a single
01292   // Number symbol, eg '123' = int (123). Scope of j needs
01293   // to be outside of the for loop.
01294 
01295   ::std::string::size_type j = 1;
01296 
01297   for (; i + j <= input.length () && is_alphanumeric (input[i + j]); ++j)
01298     continue;
01299 
01300   // make a Number out of the integer
01301 
01302   Variable * variable = new Variable (input.substr (i,j), context);
01303   variable->add_precedence (accumulated_precedence);
01304 
01305   lastValidInput = variable;
01306 
01307   // update i to next char for main loop or handle parenthesis.
01308 
01309   i += j;
01310 
01311   precedence_insert (variable, list);
01312 }

01319         Madara::Expression_Tree::Symbol *& lastValidInput)
01320 {
01321   // merge all consecutive number chars into a single Number symbol,
01322   // eg '123' = int (123). Scope of j needs to be outside of the for

01324 
01325   ::std::string::size_type j = 1;
01326

01343 // inserts a multiplication or division into the parse tree
01344 void 
01345 Madara::Expression_Tree::Interpreter::precedence_insert (
01346   Madara::Expression_Tree::Symbol *op, ::std::list<Madara::Expression_Tree::Symbol *>& list)
01347 {
01348   if (!list.empty ())
01349   {
01350     // if last element was a number, then make that our left_
01351 
01352     Symbol *parent = list.back ();
01353     Symbol *child = 0;
01354 
01355     // check to see if op is an assignment or implication, which are treated
01356     // uniquely
01357     Assignment * op_assignment = dynamic_cast <Assignment *> (op);
01358     Implies * op_implies = dynamic_cast <Implies *> (op);
01359     Unary_Operator * op_unary = dynamic_cast <Unary_Operator *> (op);
01360     
01361     // move down the right side of the tree until we find either a null or
01362     // a precedence that is >= this operation's precedence. This puts us
01363     // in the situation that we know our op should be performed after child
01364     // or child should be null (assignment or implication not withstanding)
01365     for ( child = parent->right_; 
01366         child && child->precedence () < op->precedence ();
01367         child = child->right_)
01368         parent = child;
01369 
01370     // parent->precedence is < op->precedence at this point
01371 
01372     if (op_assignment || op_implies || op_unary)
01373     {
01374       // if we are an assignment, implies, or unary op, we actually
01375       // need this op to have slightly higher precedence (it needs to be
01376       // evaluated first). This situation is signified by something like this
01377       // var1 = var2 = var3 = 1. In the list, will be var1 = var2 = var3, so parent will be
01378       // and assignment, and parent left will be var1, right and child will be assignment
01379       // and it will have a left of var2
01380       // We need to push our current assignment a bit further down, to make our pointers look
01381       // like this:
01382       //                  =                <== current parent
01383       //               /      \
01384       //             var1        =               <== current child
01385       //                      /      \
01386       //                    var2     null
01387       // What we want
01388       //                  =               
01389       //               /      \
01390       //             var1        =               <== parent
01391       //                      /      \
01392       //                    var2     null          <== child
01393       //
01394 
01395       for ( child = parent->right_; 
01396         child && child->precedence () <= op->precedence ();
01397         child = child->right_)
01398         parent = child;
01399     }
01400 
01401     // Now that we have our parent and child setup appropriately according
01402     // to precedence relationships, we should be able to modify or replace
01403     // the tree in the list
01404 
01405     if (parent->precedence() <= op->precedence ())
01406     {
01407       // this op needs to be evaluated before the parent
01408 
01409       // first let's do some error checking, so we can give helpful compile errors
01410       Operator * parent_binary = dynamic_cast <Operator *> (parent);
01411       Unary_Operator * parent_unary = dynamic_cast <Unary_Operator *> (parent);
01412 
01413       // if the parent is a binary (like addition or &&), then it needs a left hand side
01414       // if it doesn't have this, let's report a warning and insert a Leaf Node with a value
01415       // of zero
01416       if (parent_binary && !parent->left_)
01417       {
01418         // try to give as specific message as possible
01419         Both * parent_both = dynamic_cast <Both *> (parent);
01420         if (parent_both)
01421         {
01422           MADARA_DEBUG (MADARA_LOG_WARNING, (LM_TRACE, 
01423             DLINFO "KARL COMPILE WARNING: " \
01424             "Empty statements between ';' may cause slower execution.\n" \
01425             "  Attempting to prune away the extra statement.\n"));
01426         }
01427         else
01428         {
01429           MADARA_DEBUG (MADARA_LOG_WARNING, (LM_TRACE, 
01430             DLINFO "KARL COMPILE WARNING: " \
01431             "Binary operation has no left child. Inserting a zero.\n"));
01432           parent->left_ = new Number (0);
01433         }
01434       }
01435 
01436       // if the parent is a unary operator (like negate or not), then it should
01437       // NOT have a left child. This would only happen if someone input 
01438       // something like 5 ! 3, which has no meaning. This is a compile error.
01439       if (parent_unary && parent->left_)
01440       {
01441         MADARA_ERROR (MADARA_LOG_TERMINAL_ERROR, (LM_ERROR, DLINFO
01442           "\nKARL COMPILE ERROR: " \
01443             "Unary operation shouldn't have a left child.\n"));
01444         exit (-1);
01445       }
01446 
01447       // if we've gotten to this point, then we need to 
01448       // replace the child with ourself in the tree
01449       if (child)
01450       {
01451         if (op_unary)
01452         {
01453           // This is a compile error. Unary cannot have a left
01454           // child, and that is the only way that being at this
01455           // point would make sense.
01456           MADARA_ERROR (MADARA_LOG_TERMINAL_ERROR, (LM_ERROR, DLINFO
01457             "\nKARL COMPILE ERROR: " \
01458             " unary operation shouldn't have a left child.\n"));
01459           exit (-1);
01460         }
01461         else
01462           op->left_ = child;
01463       }
01464 
01465       parent->right_ = op;
01466     }
01467     else
01468     {
01469       // we are a lower precedence than our parent, so we need to replace
01470       // the tree in the list. This typically happens with assignment, implies,
01471       // logical operations (&&, ||) and equality checks
01472 
01473       op->left_ = parent;
01474       list.pop_back ();
01475       list.push_back (op);      
01476     }
01477   }
01478   else
01479   {
01480     list.push_back (op);
01481   }
01482 }
01483 
01484 void
01485 Madara::Expression_Tree::Interpreter::main_loop (Madara::Knowledge_Engine::Thread_Safe_Context & context,
01486        const ::std::string &input, ::std::string::size_type &i,
01487        Madara::Expression_Tree::Symbol *& lastValidInput,
01488        bool & handled, int & accumulated_precedence,
01489        ::std::list<Madara::Expression_Tree::Symbol *>& list)
01490 {
01491   handled = false;
01492   if (is_number (input[i]))
01493   {
01494     handled = true;
01495     // leaf node
01496     number_insert (input, i, accumulated_precedence, 
01497       list, lastValidInput);
01498   }
01499   else if (is_alphanumeric (input[i]))
01500   {
01501     handled = true;
01502     // variable leaf node
01503     variable_insert (context, input, i, accumulated_precedence,
01504       list, lastValidInput);
01505   }
01506   else if (input[i] == '+')
01507   {
01508     handled = true;
01509     Symbol * op = 0;
01510 
01511     // is this a predecrement?
01512     if (i+1 < input.size () && input[i+1] == '+')
01513     {
01514       op = new Preincrement ();
01515       ++i;
01516     }
01517     else
01518       op = new Add ();
01519 
01520     // insert the op according to left-to-right relationships
01521     op->add_precedence (accumulated_precedence);
01522     lastValidInput = 0;
01523     precedence_insert (op, list);
01524     ++i;
01525   }
01526   else if (input[i] == '-')
01527   {
01528     handled = true;
01529     Symbol * op = 0;
01530 
01531     // is this a predecrement?
01532     if (i+1 < input.size () && input[i+1] == '-')
01533     {
01534       op = new Predecrement ();
01535       ++i;
01536     }
01537     // Negate
01538     else if (!lastValidInput)
01539       op = new Negate ();
01540     // Subtract
01541     else
01542       op = new Subtract ();
01543 
01544     // insert the op according to left-to-right relationships
01545     lastValidInput = 0;
01546     op->add_precedence (accumulated_precedence);
01547     precedence_insert (op, list);
01548     ++i;
01549   }
01550   else if (input[i] == '*')
01551   {
01552     // multiplication operation
01553     handled = true;
01554     Multiply *op = new Multiply ();
01555 
01556     // insert the op according to precedence relationships
01557     op->add_precedence (accumulated_precedence);
01558     lastValidInput = 0;
01559     precedence_insert (op, list);
01560     ++i;
01561   }
01562   else if (input[i] == '%')
01563   {
01564     // multiplication operation
01565     handled = true;
01566     Modulus *op = new Modulus ();
01567 
01568     // insert the op according to precedence relationships
01569     op->add_precedence (accumulated_precedence);
01570     lastValidInput = 0;
01571     precedence_insert (op, list);
01572     ++i;
01573   }
01574   else if (input[i] == '/')
01575   {
01576     // is this a one line comment?
01577     if (i+1 < input.size () && input[i+1] == '/')
01578     {
01579       // we have a one line comment
01580       for (; i < input.size () && input[i] != '\n'; ++i);
01581     }
01582     // is this a multi-line comment?
01583     else if (i+1 < input.size () && input[i+1] == '*')
01584     {
01585       // find the matching close
01586       std::string::size_type found = input.find ("*/",i+1);
01587 
01588       // if we were able to find the matching close,
01589       // then set i to the '/' in '*/'
01590       if (found != std::string::npos)
01591         i = found + 1;
01592 
01593       // otherwise, the user apparently wanted to
01594       // comment out the rest of the file
01595       else
01596         i = input.size ();
01597     }
01598     else
01599     {
01600       // division operation
01601       handled = true;
01602       Divide *op = new Divide ();
01603       op->add_precedence (accumulated_precedence);
01604 
01605       lastValidInput = 0;
01606 
01607       // insert the op according to precedence relationships
01608       precedence_insert (op, list);
01609     }
01610     ++i;
01611   }
01612   else if (input[i] == '=')
01613   {
01614     handled = true;
01615     // division operation
01616     Symbol * op = 0;
01617 
01618     // is this an equality?
01619     if (i+1 < input.size () && input[i+1] == '=')
01620     {
01621       op = new Equality ();
01622       op->add_precedence (accumulated_precedence);
01623 
01624       lastValidInput = 0;
01625       ++i;
01626 
01627       // insert the op according to precedence relationships
01628       precedence_insert (op, list);
01629     }
01630     // is this an implication?
01631     else if (i+1 < input.size () && input[i+1] == '>')
01632     {
01633       op = new Implies ();
01634       op->add_precedence (accumulated_precedence);
01635 
01636       lastValidInput = 0;
01637       ++i;
01638 
01639       // insert the op according to precedence relationships
01640       precedence_insert (op, list);
01641     }
01642     // must be an assignment then
01643     else
01644     {
01645       op = new Assignment ();
01646       op->add_precedence (accumulated_precedence);
01647 
01648       lastValidInput = 0;
01649 
01650       // insert the op according to precedence relationships
01651       // assignment_insert (op, list);
01652       precedence_insert (op, list);
01653     }
01654     ++i;
01655   }
01656   else if (input[i] == '!')
01657   {
01658     handled = true;
01659     Symbol * op = 0;
01660 
01661     // is this an inequality?
01662     if (i+1 < input.size () && input[i+1] == '=')
01663     {
01664       op = new Inequality ();
01665       ++i;
01666     }
01667     // must be a logical not then
01668     else
01669     {
01670       op = new Not ();
01671     }
01672 
01673     // insert the op according to precedence relationships
01674     op->add_precedence (accumulated_precedence);
01675     lastValidInput = 0;
01676     precedence_insert (op, list);
01677     ++i;
01678   }
01679   else if (input[i] == '&')
01680   {
01681     // is this a logical and?
01682     if (i+1 < input.size () && input[i+1] == '&')
01683     {
01684       handled = true;
01685       Symbol * op = new And ();
01686       ++i;
01687 
01688       // insert the op according to precedence relationships
01689       op->add_precedence (accumulated_precedence);
01690       lastValidInput = 0;
01691       precedence_insert (op, list);
01692     }
01693     else
01694     {
01695       // error. We currently don't allow logical and (A & B)
01696     }
01697     ++i;
01698   }
01699   else if (input[i] == '|')
01700   {
01701     // is this a logical and?
01702     if (i+1 < input.size () && input[i+1] == '|')
01703     {
01704       handled = true;
01705       Symbol * op = new Or ();
01706       ++i;
01707 
01708       // insert the op according to precedence relationships
01709       op->add_precedence (accumulated_precedence);
01710       lastValidInput = 0;
01711       precedence_insert (op, list);
01712     }
01713     else
01714     {
01715       // error. We don't currently support logical or
01716     }
01717     ++i;
01718   }
01719   else if (input[i] == ';')
01720   {
01721     handled = true;
01722     Symbol * op = new Both ();
01723 
01724     // insert the op according to precedence relationships
01725     op->add_precedence (accumulated_precedence);
01726     lastValidInput = 0;
01727     precedence_insert (op, list);
01728     ++i;
01729   }
01730   else if (input[i] == '<')
01731   {
01732     handled = true;
01733     Symbol * op = 0;
01734 
01735     // is this a less than or equal to operator?
01736     if (i+1 < input.size () && input[i+1] == '=')
01737     {
01738       op = new Less_Than_Equal ();
01739       ++i;
01740     }
01741     // must be a less than operator
01742     else
01743       op = new Less_Than ();
01744 
01745     // insert the op according to precedence relationships
01746     op->add_precedence (accumulated_precedence);
01747     lastValidInput = 0;
01748     precedence_insert (op, list);
01749     ++i;
01750   }
01751   else if (input[i] == '>')
01752   {
01753     handled = true;
01754     Symbol * op = 0;
01755 
01756     // is this a less than or equal to operator?
01757     if (i+1 < input.size () && input[i+1] == '=')
01758     {
01759       op = new Greater_Than_Equal ();
01760       ++i;
01761     }
01762     // must be a less than operator
01763     else
01764       op = new Greater_Than ();
01765 
01766     // insert the op according to precedence relationships
01767     op->add_precedence (accumulated_precedence);
01768     lastValidInput = 0;
01769     precedence_insert (op, list);
01770     ++i;
01771   }
01772   else if (input[i] == '(')
01773   {
01774     handled = true;
01775     ++i;
01776     handle_parenthesis (context, input, i, lastValidInput, 
01777       handled, accumulated_precedence, list);
01778   }
01779   else if (input[i] == '\t' || input[i] == ' ' 
01780         || input[i] == '\r' || input[i] == '\n')
01781   {
01782     handled = true;
01783     ++i;
01784     // skip whitespace
01785   }
01786 }

01788 void 
01789 Madara::Expression_Tree::Interpreter::handle_parenthesis (
01790   Madara::Knowledge_Engine::Thread_Safe_Context & context,
01791   const ::std::string &input, ::std::string::size_type &i,
01792   Madara::Expression_Tree::Symbol *& lastValidInput,
01793   bool & handled, int & accumulated_precedence,
01794   ::std::list<Madara::Expression_Tree::Symbol *>& master_list)
01795 {
01796   /* handle parenthesis is a lot like handling a new interpret.
01797   the difference is that we have to worry about how the calling
01798   function has its list setup */
01799 
01800   accumulated_precedence += PARENTHESIS_PRECEDENCE;
01801 
01802   ::std::list<Symbol *> list;
01803 
01804   handled = false;
01805   while (i < input.length ())
01806   {
01807     main_loop (context, input, i, lastValidInput, 
01808       handled, accumulated_precedence, list);
01809 
01810     if (input[i] == ')')
01811     {
01812       handled = true;
01813       ++i;
01814       accumulated_precedence -= PARENTHESIS_PRECEDENCE;
01815       break;
01816     }
01817   }
01818 
01819   if (master_list.size () > 0 && list.size () > 0)
01820   {
01821     Symbol * lastSymbol = master_list.back ();
01822     Operator * op = dynamic_cast <Operator *> (lastSymbol);
01823     Unary_Operator * unary = dynamic_cast <Unary_Operator *>
01824       (lastSymbol);
01825 
01826 
01827     // is it a node with 2 children?
01828     if (op)
01829     {
01830       precedence_insert (list.back (), master_list);
01831     }
01832     else if (unary)
01833       // is it a unary node (like negate)
01834     {
01835       precedence_insert (list.back (), master_list);
01836     }
01837     else
01838     {
01839       // is it a terminal node (Number)
01840       // error
01841     }
01842   }
01843   else if (list.size () > 0)
01844     master_list = list;
01845 
01846   list.clear ();
01847 }

01850 // expression tree out of the parse tree
01851 Madara::Expression_Tree::Expression_Tree 
01852 Madara::Expression_Tree::Interpreter::interpret (Madara::Knowledge_Engine::Thread_Safe_Context &context, 
01853                                                  const ::std::string &input)
01854 {
01855   // return the cached expression tree if it exists
01856   ExpressionTreeMap::const_iterator found = cache_.find (input);
01857   if (found != cache_.end ())
01858     return found->second;
01859 
01860 
01861   ::std::list<Symbol *> list;
01862   //list.clear ();
01863   Symbol * lastValidInput = 0;
01864   bool handled = false;
01865   int accumulated_precedence = 0;
01866 
01867   for (::std::string::size_type i = 0;
01868     i < input.length (); )
01869   {
01870     // we took out the loop update from the for loop
01871     // and the main_loop or handle_parenthesis call
01872     // should now take care of this.
01873     main_loop (context, input, i, lastValidInput, 
01874       handled, accumulated_precedence, list);
01875 
01876 
01877     // store last valid input symbol. this is useful to the '-' operator
01878     // and will help us determine if '-' is a subtraction or a negation
01879     //if (input[i] != ' ' && input[i] != '\n')
01880     // lastValidInput = input[i];
01881   }
01882 
01883   // if the list has an element in it, then return the back of the list.
01884   if (!list.empty ())
01885   {
01886     // Invoke a recursive Expression_Tree build starting with the root
01887     // symbol. This is an example of the builder pattern. See pg 97
01888     // in GoF book.
01889 
01890     Expression_Tree tree = Expression_Tree (list.back ()->build ());
01891 
01892     // optimize the tree
01893     tree.prune ();
01894     delete list.back ();
01895 
01896     // store this optimized tree into cached memory
01897     if (cache_.find (input) == cache_.end ())
01898       cache_[input] = tree;
01899 
01900     return tree;
01901   }
01902 
01903   // If we reach this, we didn't have any symbols.
01904   return Expression_Tree ();
01905 }
